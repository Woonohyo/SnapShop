453  private void parseJsonFeed(JSONObject jsonData) {
454  realm.beginTransaction();
455  if (jsonData != null) {
456  try {
457  totalResults = jsonData.getInt("total");
458  JSONArray feedArray = jsonData.getJSONArray("data");
459 
460  for (int i = 0; i < feedArray.length(); i++) {
461  JSONObject feedObj = (JSONObject) feedArray.get(i);
462  if (realm.where(Newsfeed.class).equalTo("pid", feedObj.getInt("pid")).findFirst() != null) {
463  realm.where(Newsfeed.class).equalTo("pid", feedObj.getInt("pid")).findFirst().removeFromRealm();
464  }
465  Newsfeed item = realm.createObject(Newsfeed.class);
466  item.setPid(feedObj.getInt("pid"));
467  item.setTitle(feedObj.getString("title"));
468  // url might be null sometimes
469  String feedUrl = feedObj.isNull("shopUrl") ? null : feedObj.getString("shopUrl");
470  item.setShopUrl(feedUrl);
471  item.setContents(feedObj.getString("contents"));
472  item.setImageUrl(feedObj.getString("imgUrl"));
473  item.setNumLike(feedObj.getInt("numLike"));
474  item.setPrice(feedObj.getString("price"));
475  item.setWriteDate(feedObj.getString("writeDate"));
476  item.setWriter(feedObj.getString("writer"));
477  item.setUserLike(feedObj.getInt("like"));
478  item.setRead(feedObj.getInt("read"));
479  }
480 
481  } catch (JSONException e) {
482  e.printStackTrace();
483  realm.commitTransaction();
484  }
485  }
486 
487  realm.where(Newsfeed.class).findAll().sort("pid", RealmResults.SORT_ORDER_DESCENDING);
488  newsfeedAdapter.notifyDataSetChanged();
