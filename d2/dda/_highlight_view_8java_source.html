131  protected void draw(Canvas canvas) {
132  canvas.save();
133  Path path = new Path();
134  outlinePaint.setStrokeWidth(outlineWidth);
135  if (!hasFocus()) {
136  outlinePaint.setColor(Color.BLACK);
137  canvas.drawRect(drawRect, outlinePaint);
138  } else {
139  Rect viewDrawingRect = new Rect();
140  viewContext.getDrawingRect(viewDrawingRect);
141 
142  path.addRect(new RectF(drawRect), Path.Direction.CW);
143  outlinePaint.setColor(highlightColor);
144 
145  if (isClipPathSupported(canvas)) {
146  canvas.clipPath(path, Region.Op.DIFFERENCE);
147  canvas.drawRect(viewDrawingRect, outsidePaint);
148  } else {
149  drawOutsideFallback(canvas);
150  }
151 
152  canvas.restore();
153  canvas.drawPath(path, outlinePaint);
154 
155  if (showThirds) {
156  drawThirds(canvas);
157  }
158 
159  if (handleMode == HandleMode.Always || (handleMode == HandleMode.Changing && modifyMode == ModifyMode.Grow)) {
160  drawHandles(canvas);
161  }
162  }

221  // Determines which edges are hit by touching at (x, y)
222  public int getHit(float x, float y) {
223  Rect r = computeLayout();
224  final float hysteresis = 20F;
225  int retval = GROW_NONE;
226 
227  // verticalCheck makes sure the position is between the top and
228  // the bottom edge (with some tolerance). Similar for horizCheck.
229  boolean verticalCheck = (y >= r.top - hysteresis) && (y < r.bottom + hysteresis);
230  boolean horizCheck = (x >= r.left - hysteresis) && (x < r.right + hysteresis);
231 
232  // Check whether the position is near some edge(s)
233  if ((Math.abs(r.left - x) < hysteresis) && verticalCheck) {
234  retval |= GROW_LEFT_EDGE;
235  }
236  if ((Math.abs(r.right - x) < hysteresis) && verticalCheck) {
237  retval |= GROW_RIGHT_EDGE;
238  }
239  if ((Math.abs(r.top - y) < hysteresis) && horizCheck) {
240  retval |= GROW_TOP_EDGE;
241  }
242  if ((Math.abs(r.bottom - y) < hysteresis) && horizCheck) {
243  retval |= GROW_BOTTOM_EDGE;
244  }
245 
246  // Not near any edge but inside the rectangle: move
247  if (retval == GROW_NONE && r.contains((int) x, (int) y)) {
248  retval = MOVE;
249  }
250  return retval;
251  }

253  // Handles motion (dx, dy) in screen space.
254  // The "edge" parameter specifies which edges the user is dragging.
255  void handleMotion(int edge, float dx, float dy) {
256  Rect r = computeLayout();
257  if (edge == MOVE) {
258  // Convert to image space before sending to moveBy()
259  moveBy(dx * (cropRect.width() / r.width()), dy * (cropRect.height() / r.height()));
260  } else {
261  if (((GROW_LEFT_EDGE | GROW_RIGHT_EDGE) & edge) == 0) {
262  dx = 0;
263  }
264 
265  if (((GROW_TOP_EDGE | GROW_BOTTOM_EDGE) & edge) == 0) {
266  dy = 0;
267  }
268 
269  // Convert to image space before sending to growBy()
270  float xDelta = dx * (cropRect.width() / r.width());
271  float yDelta = dy * (cropRect.height() / r.height());
272  growBy((((edge & GROW_LEFT_EDGE) != 0) ? -1 : 1) * xDelta, (((edge & GROW_TOP_EDGE) != 0) ? -1 : 1)
273  * yDelta);
274  }
275  }

277  // Grows the cropping rectangle by (dx, dy) in image space
278  void moveBy(float dx, float dy) {
279  Rect invalRect = new Rect(drawRect);
280 
281  cropRect.offset(dx, dy);
282 
283  // Put the cropping rectangle inside image rectangle
284  cropRect.offset(Math.max(0, imageRect.left - cropRect.left), Math.max(0, imageRect.top - cropRect.top));
285 
286  cropRect.offset(Math.min(0, imageRect.right - cropRect.right), Math.min(0, imageRect.bottom - cropRect.bottom));
287 
288  drawRect = computeLayout();
289  invalRect.union(drawRect);
290  invalRect.inset(-(int) handleRadius, -(int) handleRadius);
291  viewContext.invalidate(invalRect);
292  }

294  // Grows the cropping rectangle by (dx, dy) in image space.
295  void growBy(float dx, float dy) {
296  if (maintainAspectRatio) {
297  if (dx != 0) {
298  dy = dx / initialAspectRatio;
299  } else if (dy != 0) {
300  dx = dy * initialAspectRatio;
301  }
302  }
303 
304  // Don't let the cropping rectangle grow too fast.
305  // Grow at most half of the difference between the image rectangle and
306  // the cropping rectangle.
307  RectF r = new RectF(cropRect);
308  if (dx > 0F && r.width() + 2 * dx > imageRect.width()) {
309  dx = (imageRect.width() - r.width()) / 2F;
310  if (maintainAspectRatio) {
311  dy = dx / initialAspectRatio;
312  }
313  }
314  if (dy > 0F && r.height() + 2 * dy > imageRect.height()) {
315  dy = (imageRect.height() - r.height()) / 2F;
316  if (maintainAspectRatio) {
317  dx = dy * initialAspectRatio;
318  }
319  }
320 
321  r.inset(-dx, -dy);
322 
323  // Don't let the cropping rectangle shrink too fast
324  final float widthCap = 25F;
325  if (r.width() < widthCap) {
326  r.inset(-(widthCap - r.width()) / 2F, 0F);
327  }
328  float heightCap = maintainAspectRatio ? (widthCap / initialAspectRatio) : widthCap;
329  if (r.height() < heightCap) {
330  r.inset(0F, -(heightCap - r.height()) / 2F);
331  }
332 
333  // Put the cropping rectangle inside the image rectangle
334  if (r.left < imageRect.left) {
335  r.offset(imageRect.left - r.left, 0F);
336  } else if (r.right > imageRect.right) {
337  r.offset(-(r.right - imageRect.right), 0F);
338  }
339  if (r.top < imageRect.top) {
340  r.offset(0F, imageRect.top - r.top);
341  } else if (r.bottom > imageRect.bottom) {
342  r.offset(0F, -(r.bottom - imageRect.bottom));
343  }
344 
345  cropRect.set(r);
346  drawRect = computeLayout();
347  viewContext.invalidate();
348  }

356  // Maps the cropping rectangle from image space to screen space
