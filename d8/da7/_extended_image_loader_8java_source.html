471  private void batchResponse(String cacheKey, BatchedImageRequest request) {
472  mBatchedResponses.put(cacheKey, request);
473  // If we don't already have a batch delivery runnable in flight, make a
474  // new one.
475  // Note that this will be used to deliver responses to all callers in
476  // mBatchedResponses.
477  if (mRunnable == null) {
478  mRunnable = new Runnable() {
479  @Override
480  public void run() {
481  for (BatchedImageRequest bir : mBatchedResponses.values()) {
482  for (ImageContainer container : bir.mContainers) {
483  // If one of the callers in the batched request
484  // canceled the request
485  // after the response was received but before it was
486  // delivered,
487  // skip them.
488  if (container.mListener == null) {
489  continue;
490  }
491  if (bir.getError() == null) {
492  container.mBitmap = bir.mResponseBitmap;
493  container.mListener.onResponse(container, false);
494  } else {
495  container.mListener.onErrorResponse(bir.getError());
496  }
497  }
498  }
499  mBatchedResponses.clear();
500  mRunnable = null;
501  }
502 
503  };
504  // Post the runnable.
505  mHandler.postDelayed(mRunnable, mBatchResponseDelayMs);
506  }
