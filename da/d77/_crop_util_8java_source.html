53  public static int getExifRotation(File imageFile) {
54  if (imageFile == null)
55  return 0;
56  try {
57  ExifInterface exif = new ExifInterface(imageFile.getAbsolutePath());
58  // We only recognize a subset of orientation tag values
59  switch (exif.getAttributeInt(ExifInterface.TAG_ORIENTATION, ExifInterface.ORIENTATION_UNDEFINED)) {
60  case ExifInterface.ORIENTATION_ROTATE_90:
61  return 90;
62  case ExifInterface.ORIENTATION_ROTATE_180:
63  return 180;
64  case ExifInterface.ORIENTATION_ROTATE_270:
65  return 270;
66  default:
67  return ExifInterface.ORIENTATION_UNDEFINED;
68  }
69  } catch (IOException e) {
70  Log.e("Error getting Exif data", e);
71  return 0;
72  }
73  }

75  public static boolean copyExifRotation(File sourceFile, File destFile) {

91  public static File getFromMediaUri(ContentResolver resolver, Uri uri) {
92  if (uri == null)
93  return null;
94 
95  if (SCHEME_FILE.equals(uri.getScheme())) {
96  return new File(uri.getPath());
97  } else if (SCHEME_CONTENT.equals(uri.getScheme())) {
98  final String[] filePathColumn = { MediaStore.MediaColumns.DATA, MediaStore.MediaColumns.DISPLAY_NAME };
99  Cursor cursor = null;
100  try {
101  cursor = resolver.query(uri, filePathColumn, null, null, null);
102  if (cursor != null && cursor.moveToFirst()) {
103  final int columnIndex = (uri.toString().startsWith("content://com.google.android.gallery3d")) ? cursor
104  .getColumnIndex(MediaStore.MediaColumns.DISPLAY_NAME) : cursor
105  .getColumnIndex(MediaStore.MediaColumns.DATA);
106  // Picasa image on newer devices with Honeycomb and up
107  if (columnIndex != -1) {
108  String filePath = cursor.getString(columnIndex);
109  if (!TextUtils.isEmpty(filePath)) {
110  return new File(filePath);
111  }
112  }
113  }
114  } catch (SecurityException ignored) {
115  // Nothing we can do
116  } finally {
117  if (cursor != null)
118  cursor.close();
119  }
120  }
121  return null;
122  }

132  private static class BackgroundJob extends MonitoredActivity.LifeCycleAdapter implements Runnable {
133 
134  private final MonitoredActivity mActivity;

136  private final Runnable mJob;
137  private final Handler mHandler;
138  private final Runnable mCleanupRunner = new Runnable() {
139  public void run() {
140  mActivity.removeLifeCycleListener(BackgroundJob.this);
141  if (mDialog.getWindow() != null)

146  public BackgroundJob(MonitoredActivity activity, Runnable job, ProgressDialog dialog, Handler handler) {
147  mActivity = activity;
148  mDialog = dialog;
149  mJob = job;
150  mActivity.addLifeCycleListener(this);
151  mHandler = handler;
152  }
153 
154  public void run() {

156  mJob.run();
157  } finally {
